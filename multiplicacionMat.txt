programa matrixmultiplication;

principal() {
    var int matrix1[3][4], matrix2[4][6], result[3][6], i, j;

    para j = 0 hasta j < 4 {
        para i = 0 hasta i < 3 {
            matrix1[i][j] = i + j;
        }
    }
    
    para j = 0 hasta j < 6 {
        para i = 0 hasta i < 4 {
            matrix2[i][j] = i + j;
        }
    }

    result = matrix1 * matrix2;

    para j = 0 hasta j < 6 {
        para i = 0 hasta i < 3 {
            imprime(result[i][j]);
        }
    }
}
