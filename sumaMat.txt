programa multiplicacionMatrices;

principal() {
    var int matriz1[2][2], matriz2[2][2], resultado[2][2], i, j;
    matriz1[0][0] = 1;
    matriz1[0][1] = 2;
    matriz1[1][0] = 3;
    matriz1[1][1] = 4;
    matriz2[0][0] = 1;
    matriz2[0][1] = 2;
    matriz2[1][0] = 3;
    matriz2[1][1] = 4;
    resultado = matriz1 + matriz2; 
para j = 0 hasta j < 2 {
        para i = 0 hasta i < 2 {
            imprime(resultado[i][j]);
        }
    }
}
